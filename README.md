wget -qO- https://raw.githubusercontent.com/sky1793456/singbox/main/install.sh | sudo bash









# 安装 BBR
sb bbr install

# 查看 BBR 状态
sb bbr status

# 卸载 BBR
sb bbr uninstall

# 查看节点
sb node list

# 添加节点
sb node add

# 删除日志
sb log delete

# 设置日志等级
sb log level

# 升级管理脚本
sb update script

# 卸载全部
sb uninstall
